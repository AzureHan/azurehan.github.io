# 代码实现

递归函数方式（PHP 代码）

function quicksort(&$arr)
{
    quicksort_worker($arr, 0, count($arr) - 1);
}

function quicksort_worker(&$arr, $start, $end)
{
    if ($start >= $end) {
        return ;
    }

    $pivot = quicksort_make_pivot($arr, $start, $end);

    quicksort_worker($arr, $start, $pivot - 1);
    quicksort_worker($arr, $pivot + 1, $end);
}

function quicksort_make_pivot(&$arr, $start, $end)
{
    $pivot = $start;
    for ($i = $start; $i < $end; $i++) {
        if ($arr[$i] < $arr[$end]) {
            array_item_switch($arr, $pivot, $i);
            $pivot++;
        }
    }
    array_item_switch($arr, $pivot, $end);
    return $pivot;
}

function array_item_switch(&$arr, $x, $y)
{
    $tmp = $arr[$x];
    $arr[$x] = $arr[$y];
    $arr[$y] = $tmp;
}

非递归函数方式（PHP 代码）

function quicksort(&$arr)
{
	$tasks = [[0, count($arr) - 1]];

	while (count($tasks)) {
		$task = array_pop($tasks);
		if($task[0] >= $task[1]) {
			continue ;
		}

		$pivot = $task[0];
		for ($i = $task[0]; $i < $task[1]; $i++) { 
			if($arr[$i] < $arr[$task[1]]) {
				item_switch($arr, $pivot, $i);
				$pivot++;
			}
		}
        
        /* $task[1] 位置的值作为「中间」判断元素 */
        /* $pivot 位置及其之后的所有元素都比它小 */
        /* 所以最后 $pivot 和 $task[1] 位置的元素需要互换 */
		item_switch($arr, $pivot, $task[1]);

		array_push($tasks, [$task[0], $pivot - 1]);
		array_push($tasks, [$pivot + 1, $task[1]]);
	}
}

function item_switch(&$arr, $x, $y)
{
	$tmp = $arr[$x];
	$arr[$x] = $arr[$y];
	$arr[$y] = $tmp;
}
